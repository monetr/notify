## Try it

<Demo>
  <RapidFireDemo />

```tsx
import { SnackbarProvider, useSnackbar } from '@monetr/notify';

function App() {
  return (
    <SnackbarProvider>
      <Body />
    </SnackbarProvider>
  );
}

function Body() {
  const { enqueueSnackbar } = useSnackbar();
  return (
    <button onClick={() => enqueueSnackbar('Saved', { variant: 'success' })}>
      Save
    </button>
  );
}
```
